    {
            \captionsetup{singlelinecheck=false} % left aligned
            \begin{figure}[!htbp]
                \thisfloatpagestyle{empty}
                \centering
                \begin{measuredfigure}
                    \includegraphics[width=#1\textwidth,height=#2\textheight, keepaspectratio]{#3}
                \end{measuredfigure}
                \caption{#4}
                \label{#5}
            \end{figure}
        }
}
